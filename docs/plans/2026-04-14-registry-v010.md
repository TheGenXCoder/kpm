**Goal:** Add a secrets registry to KPM — `kpm add`, `kpm list`, `kpm describe`, `kpm history`, `kpm remove` — backed by new AgentKMS write/list/delete endpoints with separated metadata and versioning.

**Architecture:** Two-repo change. AgentKMS gets new API endpoints (POST/DELETE/list for secrets and metadata) using the existing EncryptedKV store with logical path separation (`kv/secrets/` vs `kv/metadata/`). KPM gets new CLI commands that call these endpoints through the existing mTLS client. Every write is audited and policy-checked.

**Tech Stack:** Go, existing AgentKMS patterns (audit, policy, mTLS middleware), `golang.org/x/term` for masked input in KPM.

## Scope

**In scope (v0.1.0):**
- AgentKMS: POST /secrets/{path}, POST /metadata/{path}, GET /metadata, GET /metadata/{path}, DELETE /secrets/{path}, GET /secrets/{path}/history
- KPM: kpm add (interactive/pipe/file), kpm list, kpm describe, kpm history, kpm remove
- Separated storage: kv/secrets/* for values, kv/metadata/* for metadata
- Secret versioning (last 10 versions retained)
- Soft-delete with optional --purge
- Auto-detect secret type from value format
- Template resolver shorthand ${kms:service/name}

**Out of scope (v0.2.0+):**
- Import scanner (kpm import --scan)
- Templatize (kpm import --templatize)
- ABAC attributes beyond mTLS cert (time, network, tags)
- Config profiles

// ── Request/response types ──────────────────────────────────────────

type writeSecretRequest struct {
	Value            string            `json:"value,omitempty"`

}

type writeSecretResponse struct {

	Version int    `json:"version"`

}

type writeMetadataRequest struct {
	Description string   `json:"description,omitempty"`
	Tags        []string `json:"tags,omitempty"`
	Type        string   `json:"type,omitempty"`
	Expires     string   `json:"expires,omitempty"`
}

type metadataRecord struct {
	Service     string           `json:"service"`
	Name        string           `json:"name"`
	Description string           `json:"description,omitempty"`
	Tags        []string         `json:"tags,omitempty"`
	Type        string           `json:"type,omitempty"`
	Created     string           `json:"created"`
	Updated     string           `json:"updated"`
	Expires     string           `json:"expires,omitempty"`
	Version     int              `json:"version"`
	Deleted     bool             `json:"deleted"`
	Versions    []versionEntry   `json:"versions"`
}

type versionEntry struct {

	Caller  string `json:"caller"`
}

type listMetadataResponse struct {
	Secrets []metadataRecord `json:"secrets"`
}

// ── POST /secrets/{path} ────────────────────────────────────────────

func (s *Server) handleWriteSecret(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	path := r.PathValue("path")
	id := identityFromContext(ctx)

	ev, evErr := audit.New()
	if evErr != nil {
		s.writeError(w, http.StatusInternalServerError, errCodeInternal, "internal error")
		return
	}
	ev.Operation = audit.OperationSecretWrite
	ev.Environment = s.env
	ev.SourceIP = extractRemoteIP(r)
	ev.UserAgent = r.UserAgent()
	ev.CallerID = id.CallerID
	ev.TeamID = id.TeamID
	ev.KeyID = "secrets/" + path

	if path == "" {
		ev.Outcome = audit.OutcomeDenied
		ev.DenyReason = "empty path"
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusBadRequest, errCodeInvalidRequest, "path required")
		return
	}

	if s.registryWriter == nil {
		ev.Outcome = audit.OutcomeError
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusServiceUnavailable, errCodeInternal, "registry not configured")
		return
	}

	// Policy check
	decision, pErr := s.policy.Evaluate(ctx, id, audit.OperationSecretWrite, "secrets/"+path)
	if pErr != nil || !decision.Allow {
		ev.Outcome = audit.OutcomeDenied
		if pErr == nil {
			ev.DenyReason = decision.DenyReason
		}
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusForbidden, errCodePolicyDenied, "denied")
		return
	}

	// Parse request body
	r.Body = http.MaxBytesReader(w, r.Body, maxRequestBodyBytes)
	var req writeSecretRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		ev.Outcome = audit.OutcomeDenied
		ev.DenyReason = "malformed request"
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusBadRequest, errCodeInvalidRequest, "malformed request body")
		return
	}

	// Build fields map
	fields := req.Fields
	if fields == nil {
		fields = make(map[string]string)
	}
	if req.Value != "" {
		fields["value"] = req.Value
	}
	if len(fields) == 0 {
		ev.Outcome = audit.OutcomeDenied
		ev.DenyReason = "no value provided"
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusBadRequest, errCodeInvalidRequest, "value or fields required")
		return
	}

	secretsPath := "kv/data/secrets/" + path

	// Versioning: read current, copy to v{N} if exists
	version := 1
	metaPath := "kv/data/metadata/" + path
	existingMeta, _ := s.registryWriter.GetSecret(ctx, metaPath)
	if existingMeta != nil {
		if vStr, ok := existingMeta["version"]; ok {
			fmt.Sscanf(vStr, "%d", &version)
		}
		// Copy current to versioned path
		existing, getErr := s.registryWriter.GetSecret(ctx, secretsPath)
		if getErr == nil && existing != nil {
			versionedPath := fmt.Sprintf("%s/v%d", secretsPath, version)
			_ = s.registryWriter.SetSecret(ctx, versionedPath, existing)
		}
		version++
	}

	// Write new secret value
	if err := s.registryWriter.SetSecret(ctx, secretsPath, fields); err != nil {
		ev.Outcome = audit.OutcomeError
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusInternalServerError, errCodeInternal, "write failed")
		return
	}

	// Update metadata
	now := time.Now().UTC().Format(time.RFC3339)
	parts := strings.SplitN(path, "/", 2)
	service := parts[0]
	name := path
	if len(parts) == 2 {
		name = parts[1]
	}

	// Load existing metadata or create new
	meta := map[string]string{
		"service": service,
		"name":    name,
		"created": now,
		"updated": now,
		"version": fmt.Sprintf("%d", version),
		"deleted": "false",
	}
	if existingMeta != nil {
		// Preserve existing fields
		if c, ok := existingMeta["created"]; ok {
			meta["created"] = c
		}
		if d, ok := existingMeta["description"]; ok {
			meta["description"] = d
		}
		if t, ok := existingMeta["tags"]; ok {
			meta["tags"] = t
		}
		if tp, ok := existingMeta["type"]; ok {
			meta["type"] = tp
		}
		if e, ok := existingMeta["expires"]; ok {
			meta["expires"] = e
		}
		// Append to versions history
		if vh, ok := existingMeta["versions_history"]; ok {
			meta["versions_history"] = vh + ";" + fmt.Sprintf("%d,%s,%s", version, now, id.CallerID)
		} else {
			meta["versions_history"] = fmt.Sprintf("%d,%s,%s", version, now, id.CallerID)
		}
	} else {
		meta["versions_history"] = fmt.Sprintf("%d,%s,%s", version, now, id.CallerID)
	}

	if err := s.registryWriter.SetSecret(ctx, metaPath, meta); err != nil {
		ev.Outcome = audit.OutcomeError
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusInternalServerError, errCodeInternal, "metadata write failed")
		return
	}

	ev.Outcome = audit.OutcomeSuccess
	ev.KeyVersion = version
	_ = s.auditLog(ctx, ev)

	status := "created"
	if version > 1 {
		status = "updated"
	}
	writeJSON(w, http.StatusOK, writeSecretResponse{
		Path:    path,
		Version: version,
		Status:  status,
	})
}

func (s *Server) handleListMetadata(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	id := identityFromContext(ctx)

	ev, _ := audit.New()
	ev.Operation = audit.OperationMetadataList
	ev.Environment = s.env
	ev.CallerID = id.CallerID
	ev.TeamID = id.TeamID

	if s.registryWriter == nil {
		ev.Outcome = audit.OutcomeError
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusServiceUnavailable, errCodeInternal, "registry not configured")
		return
	}

	decision, pErr := s.policy.Evaluate(ctx, id, audit.OperationMetadataList, "metadata/*")
	if pErr != nil || !decision.Allow {
		ev.Outcome = audit.OutcomeDenied
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusForbidden, errCodePolicyDenied, "denied")
		return
	}

	paths, err := s.registryWriter.ListPaths(ctx)
	if err != nil {
		ev.Outcome = audit.OutcomeError
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusInternalServerError, errCodeInternal, "list failed")
		return
	}

	// Filter to metadata paths only, build records
	var records []metadataRecord
	prefix := "kv/data/metadata/"
	for _, p := range paths {
		if !strings.HasPrefix(p, prefix) {
			continue
		}
		meta, err := s.registryWriter.GetSecret(ctx, p)
		if err != nil || meta == nil {
			continue
		}
		// Skip deleted unless requested
		if meta["deleted"] == "true" {
			if r.URL.Query().Get("include_deleted") != "true" {
				continue
			}
		}

		version := 1
		if v, ok := meta["version"]; ok {
			fmt.Sscanf(v, "%d", &version)
		}

		var tags []string
		if t, ok := meta["tags"]; ok && t != "" {
			tags = strings.Split(t, ",")
		}

		expired := false
		if e, ok := meta["expires"]; ok && e != "" {
			if t, err := time.Parse(time.RFC3339, e); err == nil {
				expired = time.Now().After(t)
			}
		}

		secretPath := strings.TrimPrefix(p, prefix)
		records = append(records, metadataRecord{
			Service:     meta["service"],
			Name:        meta["name"],
			Description: meta["description"],
			Tags:        tags,
			Type:        meta["type"],
			Created:     meta["created"],
			Updated:     meta["updated"],
			Expires:     meta["expires"],
			Version:     version,
			Deleted:     meta["deleted"] == "true",
		})
		_ = secretPath // used for constructing the record
	}

	ev.Outcome = audit.OutcomeSuccess
	_ = s.auditLog(ctx, ev)

	writeJSON(w, http.StatusOK, listMetadataResponse{Secrets: records})
}

// DELETE /secrets/{path}
func (s *Server) handleDeleteSecret(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	path := r.PathValue("path")
	id := identityFromContext(ctx)
	purge := r.URL.Query().Get("purge") == "true"

	opName := audit.OperationSecretDelete
	if purge {
		opName = audit.OperationSecretPurge
	}

	ev, _ := audit.New()
	ev.Operation = opName
	ev.Environment = s.env
	ev.CallerID = id.CallerID
	ev.TeamID = id.TeamID
	ev.KeyID = "secrets/" + path

	if path == "" || s.registryWriter == nil {
		ev.Outcome = audit.OutcomeDenied
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusBadRequest, errCodeInvalidRequest, "path required")
		return
	}

	decision, _ := s.policy.Evaluate(ctx, id, opName, "secrets/"+path)
	if !decision.Allow {
		ev.Outcome = audit.OutcomeDenied
		_ = s.auditLog(ctx, ev)
		s.writeError(w, http.StatusForbidden, errCodePolicyDenied, "denied")
		return
	}

	metaPath := "kv/data/metadata/" + path

	if purge {
		// Hard delete: remove secrets + metadata + all versions
		secretsPath := "kv/data/secrets/" + path
		_ = s.registryWriter.DeleteSecret(ctx, secretsPath)
		_ = s.registryWriter.DeleteSecret(ctx, metaPath)
		// Delete versioned entries
		paths, _ := s.registryWriter.ListPaths(ctx)
		vPrefix := secretsPath + "/v"
		for _, p := range paths {
			if strings.HasPrefix(p, vPrefix) {
				_ = s.registryWriter.DeleteSecret(ctx, p)
			}
		}
	} else {
		// Soft delete: mark metadata as deleted
		meta, err := s.registryWriter.GetSecret(ctx, metaPath)
		if err != nil || meta == nil {
			ev.Outcome = audit.OutcomeDenied
			_ = s.auditLog(ctx, ev)
			s.writeError(w, http.StatusNotFound, errCodeKeyNotFound, "not found")
			return
		}
		meta["deleted"] = "true"
		meta["deleted_at"] = time.Now().UTC().Format(time.RFC3339)
		meta["deleted_by"] = id.CallerID
		_ = s.registryWriter.SetSecret(ctx, metaPath, meta)
	}

	ev.Outcome = audit.OutcomeSuccess
	_ = s.auditLog(ctx, ev)

	status := "deleted"
	if purge {
		status = "purged"
	}
	writeJSON(w, http.StatusOK, map[string]string{"path": path, "status": status})
}

// GET /secrets/{path}/history
func (s *Server) handleSecretHistory(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	path := r.PathValue("path")
	id := identityFromContext(ctx)

	if path == "" || s.registryWriter == nil {
		s.writeError(w, http.StatusBadRequest, errCodeInvalidRequest, "path required")
		return
	}

	decision, _ := s.policy.Evaluate(ctx, id, audit.OperationSecretHistory, "secrets/"+path)
	if !decision.Allow {
		s.writeError(w, http.StatusForbidden, errCodePolicyDenied, "denied")
		return
	}

	metaPath := "kv/data/metadata/" + path
	meta, err := s.registryWriter.GetSecret(ctx, metaPath)
	if err != nil || meta == nil {
		s.writeError(w, http.StatusNotFound, errCodeKeyNotFound, "not found")
		return
	}

	// Parse versions history
	var versions []versionEntry
	if vh, ok := meta["versions_history"]; ok && vh != "" {
		for _, entry := range strings.Split(vh, ";") {
			parts := strings.SplitN(entry, ",", 3)
			if len(parts) == 3 {
				v := 0
				fmt.Sscanf(parts[0], "%d", &v)
				versions = append(versions, versionEntry{
					Version: v,
					Created: parts[1],
					Caller:  parts[2],
				})
			}
		}
	}

	writeJSON(w, http.StatusOK, map[string]any{

package kpm

import "strings"

// DetectSecretType guesses the type of a secret from its value.
func DetectSecretType(value string) string {
	// API token prefixes
	apiPrefixes := []string{"sk-", "pk_", "ghp_", "gho_", "ghs_", "AKIA", "sg-", "xoxb-", "xoxp-", "Bearer "}
	for _, p := range apiPrefixes {
		if strings.HasPrefix(value, p) {
			return "api-token"
		}
	}

	// SSH keys
	if strings.Contains(value, "-----BEGIN OPENSSH PRIVATE KEY-----") ||
		strings.Contains(value, "-----BEGIN RSA PRIVATE KEY-----") ||
		strings.Contains(value, "-----BEGIN EC PRIVATE KEY-----") ||
		strings.Contains(value, "-----BEGIN PRIVATE KEY-----") {
		return "ssh-key"
	}

	// Certificates
	if strings.Contains(value, "-----BEGIN CERTIFICATE-----") {
		return "certificate"
	}

	// Connection strings
	connPrefixes := []string{"postgres://", "postgresql://", "mongodb://", "mysql://", "redis://", "amqp://"}
	for _, p := range connPrefixes {
		if strings.HasPrefix(value, p) {
			return "connection-string"
		}
	}

	// JWTs
	if strings.HasPrefix(value, "eyJ") && strings.Count(value, ".") >= 2 && len(value) > 40 {
		return "jwt"
	}

	return "generic"
}

package kpm

import (
	"bufio"
	"context"
	"fmt"
	"io"
	"os"
	"strings"

	"golang.org/x/term"
)

// AddOptions configures the kpm add command.
type AddOptions struct {
	Path        string   // service/name
	FromFile    string   // --from file path
	Description string   // --description
	Tags        []string // --tags
	Type        string   // --type (auto-detected if empty)
	Expires     string   // --expires
}

// RunAdd executes the kpm add command.
func RunAdd(ctx context.Context, w io.Writer, client *Client, opts AddOptions) error {
	if opts.Path == "" {
		return fmt.Errorf("path required (format: service/name)")
	}
	if !strings.Contains(opts.Path, "/") {
		return fmt.Errorf("path must be service/name (e.g. cloudflare/dns-token)")
	}

	// Check if secret already exists
	existing, _ := client.GetMetadata(ctx, opts.Path)
	if existing != nil && !existing.Deleted {
		fmt.Fprintf(w, "%s exists (v%d). Update to v%d? [y/N] ", opts.Path, existing.Version, existing.Version+1)
		reader := bufio.NewReader(os.Stdin)
		answer, _ := reader.ReadString('\n')
		answer = strings.TrimSpace(strings.ToLower(answer))
		if answer != "y" && answer != "yes" {
			return fmt.Errorf("cancelled")
		}
	}

	// Read the secret value
	var value []byte
	var err error

	if opts.FromFile != "" {
		// From file
		value, err = os.ReadFile(opts.FromFile)
		if err != nil {
			return fmt.Errorf("read file %s: %w", opts.FromFile, err)
		}
	} else if !term.IsTerminal(int(os.Stdin.Fd())) {
		// From pipe
		value, err = io.ReadAll(os.Stdin)
		if err != nil {
			return fmt.Errorf("read stdin: %w", err)
		}
		value = []byte(strings.TrimRight(string(value), "\n"))
	} else {
		// Interactive — masked input
		fmt.Fprint(w, "Value: ")
		value, err = term.ReadPassword(int(os.Stdin.Fd()))
		fmt.Fprintln(w) // newline after masked input
		if err != nil {
			return fmt.Errorf("read value: %w", err)
		}
	}
	defer ZeroBytes(value)

	if len(value) == 0 {
		return fmt.Errorf("empty value")
	}

	// Auto-detect type
	secretType := opts.Type
	if secretType == "" {
		secretType = DetectSecretType(string(value))
	}

	// Write secret
	result, err := client.WriteSecret(ctx, opts.Path, value)
	if err != nil {
		return fmt.Errorf("store secret: %w", err)
	}

	// Write metadata
	if opts.Description != "" || opts.Tags != nil || secretType != "generic" || opts.Expires != "" {
		if err := client.WriteMetadata(ctx, opts.Path, opts.Description, opts.Tags, secretType, opts.Expires); err != nil {
			fmt.Fprintf(w, "warning: secret stored but metadata update failed: %v\n", err)
		}
	}

	tagStr := ""
	if len(opts.Tags) > 0 {
		tagStr = fmt.Sprintf(" (tagged: %s)", strings.Join(opts.Tags, ", "))
	}
	fmt.Fprintf(w, "Stored %s v%d%s [%s]\n", result.Path, result.Version, tagStr, secretType)
	return nil
}

// RunList executes the kpm list command.
func RunList(ctx context.Context, w io.Writer, client *Client, service, tag, secretType string, includeDeleted bool) error {
	secrets, err := client.ListMetadata(ctx, includeDeleted)
	if err != nil {
		return err
	}

	if len(secrets) == 0 {
		fmt.Fprintln(w, "No secrets found. Add one with: kpm add service/name")
		return nil
	}

	// Filter
	var filtered []SecretMetadata
	for _, s := range secrets {
		if service != "" && s.Service != service {
			continue
		}
		if tag != "" && !containsTag(s.Tags, tag) {
			continue
		}
		if secretType != "" && s.Type != secretType {
			continue
		}
		filtered = append(filtered, s)
	}

	if len(filtered) == 0 {
		fmt.Fprintln(w, "No secrets match the filter.")
		return nil
	}

	// Group by service
	grouped := map[string][]SecretMetadata{}
	for _, s := range filtered {
		grouped[s.Service] = append(grouped[s.Service], s)
	}

	// Sort services
	services := make([]string, 0, len(grouped))
	for k := range grouped {
		services = append(services, k)
	}
	sort.Strings(services)

	for _, svc := range services {
		fmt.Fprintf(w, "%s/\n", svc)
		for _, s := range grouped[svc] {
			typeStr := s.Type
			if typeStr == "" {
				typeStr = "generic"
			}

			tagStr := ""
			if len(s.Tags) > 0 {
				tagStr = fmt.Sprintf("[%s]", strings.Join(s.Tags, ", "))
			}

			descStr := ""
			if s.Description != "" {
				descStr = fmt.Sprintf("%q", s.Description)
			}

			vStr := fmt.Sprintf("v%d", s.Version)

			expiryStr := ""
			if s.Expires != "" {
				if t, err := time.Parse(time.RFC3339, s.Expires); err == nil {
					remaining := time.Until(t)
					if remaining < 0 {
						expiryStr = "EXPIRED"
					} else if remaining < 30*24*time.Hour {
						expiryStr = fmt.Sprintf("EXPIRES in %dd", int(remaining.Hours()/24))
					}
				}
			}

			deletedStr := ""
			if s.Deleted {
				deletedStr = "DELETED"
			}

			fmt.Fprintf(w, "  %-22s %-16s %-20s %-30s %s %s %s\n",
				s.Name, typeStr, tagStr, descStr, vStr, expiryStr, deletedStr)
		}
		fmt.Fprintln(w)
	}

	fmt.Fprintf(w, "%d secrets across %d services\n", len(filtered), len(services))
	return nil
}

		if t == tag {

	return nil
}

package kpm

import (
	"bufio"
	"context"
	"fmt"
	"io"
	"os"
	"strings"

func RunRemove(ctx context.Context, w io.Writer, client *Client, path string, purge bool) error {
	meta, err := client.GetMetadata(ctx, path)
	if err != nil {
		return fmt.Errorf("%s not found", path)
	}

	if purge {
		fmt.Fprintf(w, "PERMANENTLY delete %s (v%d) and all versions? This cannot be undone. [y/N] ", path, meta.Version)
	} else {
		fmt.Fprintf(w, "Remove %s (v%d)? Secret will be marked deleted. [y/N] ", path, meta.Version)
	}

	reader := bufio.NewReader(os.Stdin)
	answer, _ := reader.ReadString('\n')
	answer = strings.TrimSpace(strings.ToLower(answer))
	if answer != "y" && answer != "yes" {
		return fmt.Errorf("cancelled")
	}

	if err := client.DeleteSecret(ctx, path, purge); err != nil {
		return err
	}

	if purge {
		fmt.Fprintf(w, "Purged %s\n", path)
	} else {
		fmt.Fprintf(w, "Removed %s\n", path)
	}
	return nil
}
